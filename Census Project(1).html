# Translate what can be transalted, the rest is better left in Greek as it is mostly NAMES. Assign everything to the temp list.
temp_list=[]

for desc in newdf['Description'].values:
    desc=desc.split()
    if desc[0]== '':
        desc[0]=desc[0].replace('','Municipality')
    elif desc[0]== '' and desc[1]=='':
        desc[0]=desc[0].replace('','Municipal')
        desc[1]=desc[1].replace('','Unit')
    elif desc[0]== '':
        desc[0]=desc[0].replace('','Regional')
        desc[1]=desc[1].replace('','Unit')
    elif desc[0]== '':
        desc[0]=desc[0].replace('','Region')
    elif desc[0]== '':
        desc[0]=desc[0].replace('','Decentralized')
        desc[1]=desc[1].replace('','Administration')
    elif desc[0]=='':
        desc[0]=desc[0].replace('','Municipal')
        desc[1]=desc[1].replace('','Units')
    elif '()' in desc:
        desc[-1]=desc[-1].replace('()', 'Self-Governing')
    if '' in desc:
        temp_index=desc.index('')
        desc[temp_index]=desc[temp_index].replace('', 'And')
    temp_list.append(' '.join(desc))
    
newdf['Geographic Key'][0]= 0
newdf['Description'].values[0]= 'Country Total'
#replace the descriptions with the translated 
newdf['Description']= temp_list

import numpy as np

# Sample data
categories = ['Patras', 'Thessaloniki', 'Athens']
males = [males_patras, males_thesalloniki, males_athens]
females = [females_patras, females_thesalloniki, females_athens]
# Create an array for the x-axis positions
x = np.arange(len(categories))

# Plotting the stacked bar chart for males and females in each population
fig, ax = plt.subplots()

# Plotting females
ax.bar(x, females, color='red', label='Females')

# Plotting males on top of females
ax.bar(x, males, bottom=females, color='blue', label='Males')
percentage_of_males=[] #initialize a list to count the percentages
for i in range(len(categories)):
    total = males[i] + females[i]
    percentage_of_males.append(round(males[i]/total*100,2))
    ax.text(x[i], total/2*1.2, str(percentage_of_males[i]), ha='center', va='center', color='white', fontweight='bold')

labels=['Female','Male']
handles= ax.get_legend_handles_labels()[0]

plt.xlabel('Municipalities')
plt.ylabel('Total Population')
plt.title('Population Distribution (Males and Females) Across Different Municipalities')
plt.xticks(x, categories)
plt.legend(reversed(handles), reversed(labels), title="Legend", loc='upper left', bbox_to_anchor=(1, 1))
plt.show()
